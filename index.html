Apps Script 연결

배포된 웹앱 URL을 붙여넣고 저장하세요. (jinbody 때랑 같은 방식)

처음 설정하는 방법

  1. 구글 시트 새로 만들기 → 시트(탭) 이름을 매물로 변경
  2. 확장 프로그램 → Apps Script → 아래 코드 전체 붙여넣기
  3. 배포 → 새 배포 → 유형: 웹 앱 / 실행: 나 / 액세스: 모든 사용자
  4. 나온 URL을 위 칸에 붙여넣고 저장

헤더(1행)는 첫 저장 때 자동으로 만들어집니다.

const SHEET_NAME = '매물'; const HEADERS = ['매물번호','등록일','상태','주소','단지건물명','동호','거래유형','매매가','보증금','월세','전용면적','공급면적','대지면적','연면적','방수','욕실수','해당층','총층','방향','엘리베이터','주차','건폐율용적률','위반건축물','사용승인일','관리비','입주가능일','입주확정여부','공시가격','공시가기준연도','집주인연락처','세입자연락처','비고']; function getSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET_NAME); if (!sh) sh = ss.insertSheet(SHEET_NAME); if (sh.getLastRow() === 0) sh.appendRow(HEADERS); return sh; } function doGet(e) { const sh = getSheet_(); const data = sh.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).map(r => { const o = {}; headers.forEach((h, i) => o[h] = r[i]); return o; }); return ContentService.createTextOutput(JSON.stringify({ ok: true, rows })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sh = getSheet_(); const body = JSON.parse(e.postData.contents); const headers = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0]; if (body.action === 'add') { const nums = sh.getLastRow() > 1 ? sh.getRange(2, 1, sh.getLastRow() - 1, 1).getValues().flat() : []; let max = 0; nums.forEach(v => { const m = String(v).match(/^M(\d+)$/); if (m) max = Math.max(max, parseInt(m[1], 10)); }); const id = 'M' + String(max + 1).padStart(3, '0'); body.data['매물번호'] = id; body.data['등록일'] = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM-dd'); sh.appendRow(headers.map(h => body.data[h] !== undefined ? body.data[h] : '')); return ContentService.createTextOutput(JSON.stringify({ ok: true, id })) .setMimeType(ContentService.MimeType.JSON); } if (body.action === 'update') { const ids = sh.getRange(2, 1, Math.max(sh.getLastRow() - 1, 1), 1).getValues().flat(); const idx = ids.indexOf(body['매물번호']); if (idx === -1) return ContentService.createTextOutput(JSON.stringify({ ok: false, error: 'not found' })) .setMimeType(ContentService.MimeType.JSON); const row = idx + 2; Object.keys(body.data).forEach(k => { const c = headers.indexOf(k); if (c !== -1) sh.getRange(row, c + 1).setValue(body.data[k]); }); return ContentService.createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ ok: false, error: 'unknown action' })) .setMimeType(ContentService.MimeType.JSON); }
